reverse()

將陣列元素順序反轉，會改變原始陣列

let arrName = ['柏叡','騰寓','胤辰']

let arrNameReversed = arrNum.reverse()

sort([compareFunction(a,b)])

將陣列元素排序，會改變原始陣列
  • 預設排序：會以 Unicode 編碼位置來排序，也就是照字母排序
  • 自定義排序：
      • 加上 [compareFunction(a,b)] 自定義比較函式，函式中應包含兩個參數 a 與 b，
      透過比較 a 與 b 然後傳回一個數字，可能是正數、負數或 0 ，並得出以下結果：
    • 回傳值 > 0 (正數)，表示 b 排序在 a 前面
    • 回傳值 < 0 (負數)，表示 a 排序在 b 前面
    • 回傳值 = 0，不執行任何結果
> 預設排序

>自定義排序

let arrNumf = [40, 1, 5, 200]
let arrStringNum = ["80", "9", "700"]
let arrMixed = ["80", "9", "700", 40, 1, 5, 200]

function compareNumbers(a, b) {
return a - b;
}

arrNumf.sort()
arrNumf.sort(compareNumbers)
arrStringNum.sort()
arrStringNum.sort(compareNumbers)
arrMixed.sort()
arrMixed.sort(compareNumbers)
>應用在陣列物件

let MFEE40 = [
{ name: "Bread", age: 25 },
{ name: "Steven", age: 30 },
{ name: "Tom", age: 26 },
{ name: "Fix", age: 28 },
{ name: "Mei", age: 18 },
{ name: "Bear", age: 33 },
]

排序 age 數值
MFEE40.sort((a, b) => a.age - b.age)
排序 name 字串
MFEE40.sort((a, b) => {
let nameA = a.name.toUpperCase();
let nameB = b.name.toUpperCase();
if (nameA < nameB) {
return -1;
}
if (nameA > nameB) {
return 1;
}
return 0;
})

用法同reverse()，但不會修改原陣列
